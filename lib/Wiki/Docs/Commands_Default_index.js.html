function caseChecker(
    commandName = '',
    onlyCommand = false,
    isCommand = false,
) {
    /* Checa se possui o comando, nÃ£o possui case insensitive */
    if (
        (commandFound.includes(commandName) && onlyCommand === false && isCommand === false)
        || (commandFound === commandName && onlyCommand === true && isCommand === true)

        return commandFound;
    }

    /* Retorna que nÃ£o achou */
    return false;
